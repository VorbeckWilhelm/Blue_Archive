scripted_gui = {
	country_intro_window ={
		window_name = "country_intro_window"
		context_type = player_context
		
		visible = {
			ROOT = {
				NOT = {
					has_country_flag = intro_gui_close
				}
			}
		}

		effects = { #close button
			close_button_click = {
				set_country_flag = intro_gui_close
			}
		}

		effects = { #page one button
			sector_button_1_click = {
				set_country_flag = intro_gui_page_1
				clr_country_flag = intro_gui_page_2
				clr_country_flag = intro_gui_page_3
			}
		}
		effects = { #page two button
			sector_button_2_click = {
				set_country_flag = intro_gui_page_2
				clr_country_flag = intro_gui_page_1
				clr_country_flag = intro_gui_page_3
			}
		}
		effects = { #page three button
			sector_button_3_click = {
				set_country_flag = intro_gui_page_3
				clr_country_flag = intro_gui_page_1
				clr_country_flag = intro_gui_page_2
			}
		}
		triggers = {
			introduction_text_Temp_visible = {
				NOT = {
					has_country_flag =  intro_gui_page_1
				}
				NOT = {
					has_country_flag =  intro_gui_page_2
				}
				NOT = {
					has_country_flag =  intro_gui_page_3
				}
			}
		}
	}
}
